#include<stdio.h>
#include<stdlib.h>
void write(int x[],int n)
{
    for(int i=1;i<=n;i++)
        printf("%d\t",x[i]);
    printf("\n");
}
int place(int k,int i,int x[])
{
    for(int j=1;j<=k-1;j++)
    {
        if((x[j]==i) || abs(j-k)==abs(x[j]-i))
            return 0;
    }
    return 1;
}
void NQueens(int k,int n,int x[])
{
    for(int i=1;i<=n;i++)
    {
        if(place(k,i,x))
        {
            x[k]=i;
            if(k==n)
                write(x,n);
            else
                NQueens(k+1,n,x);
        }
    }
}
int main()
{
    int n,x[10];
    printf("Enter no. of Queens: ");
    scanf("%d",&n);
    NQueens(1,n,x);
    
    return 0;
}
